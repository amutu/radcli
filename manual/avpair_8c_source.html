61 int rc_avpair_assign (VALUE_PAIR *vp, void const *pval, int len)
62 {
63 
64  switch (vp->type)
65  {
66  case PW_TYPE_STRING:
67  if (len == -1)
68  len = (uint32_t)strlen((char const *)pval);
69  if (len > AUTH_STRING_LEN) {
70  rc_log(LOG_ERR, "rc_avpair_assign: bad attribute length");
71  return -1;
72  }
73  memcpy(vp->strvalue, (char const *)pval, len);
74  vp->strvalue[len] = '\0';
75  vp->lvalue = len;
76  break;
77 
78  case PW_TYPE_DATE:
79  case PW_TYPE_INTEGER:
80  case PW_TYPE_IPADDR:
81  vp->lvalue = * (uint32_t *) pval;
82  break;
83  case PW_TYPE_IPV6ADDR:
84  if (len != 16) {
85  rc_log(LOG_ERR, "rc_avpair_assign: bad IPv6 length");
86  return -1;
87  }
88  memcpy(vp->strvalue, (char const *)pval, len);
89  vp->lvalue = len;
90  break;
91 
92  case PW_TYPE_IPV6PREFIX:
93  if (len < 2 || len > 18) {
94  rc_log(LOG_ERR, "rc_avpair_assign: bad IPv6 prefix length");
95  return -1;
96  }
97  memcpy(vp->strvalue, (char const *)pval, len);
98  vp->lvalue = len;
99  break;
100 
101  default:
102  rc_log(LOG_ERR, "rc_avpair_assign: no attribute %d in dictionary", vp->type);
103  return -1;
104  }
105  return 0;
106 }

117 VALUE_PAIR *rc_avpair_new (rc_handle const *rh, int attrid, void const *pval, int len, int vendorpec)
118 {
119  VALUE_PAIR *vp = NULL;
120  DICT_ATTR *pda;
121 
122  attrid = attrid | (vendorpec << 16);
123  if ((pda = rc_dict_getattr (rh, attrid)) == NULL)
124  {
125  rc_log(LOG_ERR,"rc_avpair_new: no attribute %d in dictionary", attrid);
126  return NULL;
127  }
128  if (vendorpec != 0 && rc_dict_getvend(rh, vendorpec) == NULL)
129  {
130  rc_log(LOG_ERR,"rc_avpair_new: no Vendor-Id %d in dictionary", vendorpec);
131  return NULL;
132  }
133  if ((vp = malloc (sizeof (VALUE_PAIR))) != NULL)
134  {
135  strlcpy (vp->name, pda->name, sizeof (vp->name));
136  vp->attribute = attrid;
137  vp->next = NULL;
138  vp->type = pda->type;
139  if (rc_avpair_assign (vp, pval, len) == 0)
140  {
141  /* XXX: Fix up Digest-Attributes */
142  switch (vp->attribute) {
143  case PW_DIGEST_REALM:
144  case PW_DIGEST_NONCE:
145  case PW_DIGEST_METHOD:
146  case PW_DIGEST_URI:
147  case PW_DIGEST_QOP:
148  case PW_DIGEST_ALGORITHM:
149  case PW_DIGEST_BODY_DIGEST:
150  case PW_DIGEST_CNONCE:
151  case PW_DIGEST_NONCE_COUNT:
152  case PW_DIGEST_USER_NAME:
153  /* overlapping! */
154  if (vp->lvalue > AUTH_STRING_LEN - 2)
155  vp->lvalue = AUTH_STRING_LEN - 2;
156  memmove(&vp->strvalue[2], &vp->strvalue[0], vp->lvalue);
157  vp->strvalue[0] = vp->attribute - PW_DIGEST_REALM + 1;
158  vp->lvalue += 2;
159  vp->strvalue[1] = vp->lvalue;
160  vp->strvalue[vp->lvalue] = '\0';
161  vp->attribute = PW_DIGEST_ATTRIBUTES;
162  default:
163  break;
164  }
165  return vp;
166  }
167  free (vp);
168  vp = NULL;
169  }
170  else
171  {
172  rc_log(LOG_CRIT,"rc_avpair_new: out of memory");
173  }
174 
175  return vp;
176 }

189 VALUE_PAIR *rc_avpair_gen(rc_handle const *rh, VALUE_PAIR *pair, unsigned char const *ptr,
190  int length, int vendorpec)
191 {
192  int attribute, attrlen, x_len;
193  unsigned char const *x_ptr;
194  uint32_t lvalue;
195  DICT_ATTR *attr;
196  VALUE_PAIR *rpair;
197  char buffer[(AUTH_STRING_LEN * 2) + 1];
198  /* For hex string conversion. */
199  char hex[3];
200 
201  if (length < 2) {
202  rc_log(LOG_ERR, "rc_avpair_gen: received attribute with "
203  "invalid length");
204  goto shithappens;
205  }
206  attrlen = ptr[1];
207  if (length < attrlen || attrlen < 2) {
208  rc_log(LOG_ERR, "rc_avpair_gen: received attribute with "
209  "invalid length");
210  goto shithappens;
211  }
212 
213  /* Advance to the next attribute and process recursively */
214  if (length != attrlen) {
215  pair = rc_avpair_gen(rh, pair, ptr + attrlen, length - attrlen,
216  vendorpec);
217  if (pair == NULL)
218  return NULL;
219  }
220 
221  /* Actual processing */
222  attribute = ptr[0] | (vendorpec << 16);
223  ptr += 2;
224  attrlen -= 2;
225 
226  /* VSA */
227  if (attribute == PW_VENDOR_SPECIFIC) {
228  if (attrlen < 4) {
229  rc_log(LOG_ERR, "rc_avpair_gen: received VSA "
230  "attribute with invalid length");
231  goto skipit;
232  }
233  memcpy(&lvalue, ptr, 4);
234  vendorpec = ntohl(lvalue);
235  if (rc_dict_getvend(rh, vendorpec) == NULL) {
236  /* Warn and skip over the unknown VSA */
237  rc_log(LOG_WARNING, "rc_avpair_gen: received VSA "
238  "attribute with unknown Vendor-Id %d", vendorpec);
239  goto skipit;
240  }
241  /* Process recursively */
242  return rc_avpair_gen(rh, pair, ptr + 4, attrlen - 4,
243  vendorpec);
244  }
245 
246  /* Normal */
247  attr = rc_dict_getattr(rh, attribute);
248  if (attr == NULL) {
249  buffer[0] = '\0'; /* Initial length. */
250  x_ptr = ptr;
251  for (x_len = attrlen; x_len > 0; x_len--, x_ptr++) {
252  snprintf(hex, sizeof(hex), "%2.2X", x_ptr[0]);
253  strcat(buffer, hex);
254  }
255  if (vendorpec == 0) {
256  rc_log(LOG_WARNING, "rc_avpair_gen: received "
257  "unknown attribute %d of length %d: 0x%s",
258  attribute, attrlen + 2, buffer);
259  } else {
260  rc_log(LOG_WARNING, "rc_avpair_gen: received "
261  "unknown VSA attribute %d, vendor %d of "
262  "length %d: 0x%s", attribute & 0xffff,
263  VENDOR(attribute), attrlen + 2, buffer);
264  }
265  goto skipit;
266  }
267 
268  rpair = malloc(sizeof(*rpair));
269  if (rpair == NULL) {
270  rc_log(LOG_CRIT, "rc_avpair_gen: out of memory");
271  goto shithappens;
272  }
273  memset(rpair, '\0', sizeof(*rpair));
274 
275  /* Insert this new pair at the beginning of the list */
276  rpair->next = pair;
277  pair = rpair;
278  strcpy(pair->name, attr->name);
279  pair->attribute = attr->value;
280  pair->type = attr->type;
281 
282  switch (attr->type) {
283  case PW_TYPE_STRING:
284  memcpy(pair->strvalue, (char *)ptr, (size_t)attrlen);
285  pair->strvalue[attrlen] = '\0';
286  pair->lvalue = attrlen;
287  break;
288 
289  case PW_TYPE_INTEGER:
290  if (attrlen != 4) {
291  rc_log(LOG_ERR, "rc_avpair_gen: received INT "
292  "attribute with invalid length");
293  goto skipit;
294  }
295  case PW_TYPE_IPADDR:
296  if (attrlen != 4) {
297  rc_log(LOG_ERR, "rc_avpair_gen: received IPADDR"
298  " attribute with invalid length");
299  goto skipit;
300  }
301  memcpy((char *)&lvalue, (char *)ptr, 4);
302  pair->lvalue = ntohl(lvalue);
303  break;
304  case PW_TYPE_IPV6ADDR:
305  if (attrlen != 16) {
306  rc_log(LOG_ERR, "rc_avpair_gen: received IPV6ADDR"
307  " attribute with invalid length");
308  goto skipit;
309  }
310  memcpy(pair->strvalue, (char *)ptr, 16);
311  pair->lvalue = attrlen;
312  break;
313  case PW_TYPE_IPV6PREFIX:
314  if (attrlen > 18 || attrlen < 2) {
315  rc_log(LOG_ERR, "rc_avpair_gen: received IPV6PREFIX"
316  " attribute with invalid length: %d", attrlen);
317  goto skipit;
318  }
319  memcpy(pair->strvalue, (char *)ptr, attrlen);
320  pair->lvalue = attrlen;
321  break;
322  case PW_TYPE_DATE:
323  if (attrlen != 4) {
324  rc_log(LOG_ERR, "rc_avpair_gen: received DATE "
325  "attribute with invalid length");
326  goto skipit;
327  }
328 
329  default:
330  rc_log(LOG_WARNING, "rc_avpair_gen: %s has unknown type",
331  attr->name);
332  goto skipit;
333  }
334 
335 skipit:
336  return pair;
337 
338 shithappens:
339  while (pair != NULL) {
340  rpair = pair->next;
341  free(pair);
342  pair = rpair;
343  }
344  return NULL;
345 }

373 void rc_avpair_insert(VALUE_PAIR **a, VALUE_PAIR *p, VALUE_PAIR *b)
374 {
375  VALUE_PAIR *this_node = NULL;
376  VALUE_PAIR *vp;
377 
378  if (b->next != NULL)
379  {
380  rc_log(LOG_CRIT, "rc_avpair_insert: value pair (0x%p) next ptr. (0x%p) not NULL", b, b->next);
381  abort ();
382  }
383 
384  if (*a == NULL)
385  {
386  *a = b;
387  return;
388  }
389 
390  vp = *a;
391 
392  if ( p == NULL) /* run to end of "a" list */
393  {
394  while (vp != NULL)
395  {
396  this_node = vp;
397  vp = vp->next;
398  }
399  }
400  else /* look for the "p" entry in the "a" list */
401  {
402  this_node = *a;
403  while (this_node != NULL)
404  {
405  if (this_node == p)
406  {
407  break;
408  }
409  this_node = this_node->next;
410  }
411  }
412 
413  b->next = this_node->next;
414  this_node->next = b;
415 
416  return;
417 }

445 static void rc_fieldcpy(char *string, char const **uptr, char const *stopat, size_t len)
446 {
447  char const *ptr, *estring;
448 
449  ptr = *uptr;
450  estring = string + len - 1;
451  if (*ptr == '"')
452  {
453  ptr++;
454  while (*ptr != '"' && *ptr != '\0' && *ptr != '\n')
455  {
456  if (string < estring)
457  *string++ = *ptr;
458  ptr++;
459  }
460  if (*ptr == '"')
461  {
462  ptr++;
463  }
464  *string = '\0';
465  *uptr = ptr;
466  return;
467  }
468 
469  while (*ptr != '\0' && strchr(stopat, *ptr) == NULL)
470  {
471  if (string < estring)
472  *string++ = *ptr;
473  ptr++;
474  }
475  *string = '\0';
476  *uptr = ptr;
477  return;
478 }

492 int rc_avpair_parse (rc_handle const *rh, char const *buffer, VALUE_PAIR **first_pair)
493 {
494  int mode;
495  char attrstr[AUTH_ID_LEN];
496  char valstr[AUTH_STRING_LEN + 1], *p;
497  DICT_ATTR *attr = NULL;
498  DICT_VALUE *dval;
499  VALUE_PAIR *pair;
500  VALUE_PAIR *link;
501  struct tm *tm;
502  time_t timeval;
503 
504  mode = PARSE_MODE_NAME;
505  while (*buffer != '\n' && *buffer != '\0')
506  {
507  if (*buffer == ' ' || *buffer == '\t')
508  {
509  buffer++;
510  continue;
511  }
512 
513  switch (mode)
514  {
515  case PARSE_MODE_NAME: /* Attribute Name */
516  rc_fieldcpy (attrstr, &buffer, " \t\n=,", sizeof(attrstr));
517  if ((attr =
518  rc_dict_findattr (rh, attrstr)) == NULL)
519  {
520  rc_log(LOG_ERR, "rc_avpair_parse: unknown attribute");
521  if (*first_pair) {
522  rc_avpair_free(*first_pair);
523  *first_pair = NULL;
524  }
525  return -1;
526  }
527  mode = PARSE_MODE_EQUAL;
528  break;
529 
530  case PARSE_MODE_EQUAL: /* Equal sign */
531  if (*buffer == '=')
532  {
533  mode = PARSE_MODE_VALUE;
534  buffer++;
535  }
536  else
537  {
538  rc_log(LOG_ERR, "rc_avpair_parse: missing or misplaced equal sign");
539  if (*first_pair) {
540  rc_avpair_free(*first_pair);
541  *first_pair = NULL;
542  }
543  return -1;
544  }
545  break;
546 
547  case PARSE_MODE_VALUE: /* Value */
548  rc_fieldcpy (valstr, &buffer, " \t\n,", sizeof(valstr));
549 
550  if ((pair = malloc (sizeof (VALUE_PAIR))) == NULL)
551  {
552  rc_log(LOG_CRIT, "rc_avpair_parse: out of memory");
553  if (*first_pair) {
554  rc_avpair_free(*first_pair);
555  *first_pair = NULL;
556  }
557  return -1;
558  }
559  strcpy (pair->name, attr->name);
560  pair->attribute = attr->value;
561  pair->type = attr->type;
562 
563  switch (pair->type)
564  {
565 
566  case PW_TYPE_STRING:
567  strcpy (pair->strvalue, valstr);
568  pair->lvalue = (uint32_t)strlen(valstr);
569  break;
570 
571  case PW_TYPE_INTEGER:
572  if (isdigit (*valstr))
573  {
574  pair->lvalue = atoi (valstr);
575  }
576  else
577  {
578  if ((dval = rc_dict_findval (rh, valstr))
579  == NULL)
580  {
581  rc_log(LOG_ERR, "rc_avpair_parse: unknown attribute value: %s", valstr);
582  if (*first_pair) {
583  rc_avpair_free(*first_pair);
584  *first_pair = NULL;
585  }
586  free (pair);
587  return -1;
588  }
589  else
590  {
591  pair->lvalue = dval->value;
592  }
593  }
594  break;
595 
596  case PW_TYPE_IPADDR:
597  if (inet_pton(AF_INET, valstr, &pair->lvalue) == 0) {
598  rc_log(LOG_ERR, "rc_avpair_parse: invalid IPv4 address %s", valstr);
599  free(pair);
600  return -1;
601  }
602 
603  pair->lvalue = ntohl(pair->lvalue);
604  break;
605 
606  case PW_TYPE_IPV6ADDR:
607  if (inet_pton(AF_INET6, valstr, pair->strvalue) == 0) {
608  rc_log(LOG_ERR, "rc_avpair_parse: invalid IPv6 address %s", valstr);
609  free(pair);
610  return -1;
611  }
612  pair->lvalue = 16;
613  break;
614 
615  case PW_TYPE_IPV6PREFIX:
616  p = strchr(valstr, '/');
617  if (p == NULL) {
618  rc_log(LOG_ERR, "rc_avpair_parse: invalid IPv6 prefix %s", valstr);
619  free(pair);
620  return -1;
621  }
622  *p = 0;
623  p++;
624  pair->strvalue[0] = 0;
625  pair->strvalue[1] = atoi(p);
626 
627  if (inet_pton(AF_INET6, valstr, pair->strvalue+2) == 0) {
628  rc_log(LOG_ERR, "rc_avpair_parse: invalid IPv6 prefix %s", valstr);
629  free(pair);
630  return -1;
631  }
632  pair->lvalue = 2+16;
633  break;
634 
635  case PW_TYPE_DATE:
636  timeval = time (0);
637  tm = localtime (&timeval);
638  tm->tm_hour = 0;
639  tm->tm_min = 0;
640  tm->tm_sec = 0;
641  rc_str2tm (valstr, tm);
642 #ifdef TIMELOCAL
643  pair->lvalue = (uint32_t) timelocal (tm);
644 #else /* TIMELOCAL */
645  pair->lvalue = (uint32_t) mktime (tm);
646 #endif /* TIMELOCAL */
647  break;
648 
649  default:
650  rc_log(LOG_ERR, "rc_avpair_parse: unknown attribute type %d", pair->type);
651  if (*first_pair) {
652  rc_avpair_free(*first_pair);
653  *first_pair = NULL;
654  }
655  free (pair);
656  return -1;
657  }
658 
659  /* XXX: Fix up Digest-Attributes */
660  switch (pair->attribute) {
661  case PW_DIGEST_REALM:
662  case PW_DIGEST_NONCE:
663  case PW_DIGEST_METHOD:
664  case PW_DIGEST_URI:
665  case PW_DIGEST_QOP:
666  case PW_DIGEST_ALGORITHM:
667  case PW_DIGEST_BODY_DIGEST:
668  case PW_DIGEST_CNONCE:
669  case PW_DIGEST_NONCE_COUNT:
670  case PW_DIGEST_USER_NAME:
671  /* overlapping! */
672  if (pair->lvalue > AUTH_STRING_LEN - 2)
673  pair->lvalue = AUTH_STRING_LEN - 2;
674  memmove(&pair->strvalue[2], &pair->strvalue[0], pair->lvalue);
675  pair->strvalue[0] = pair->attribute - PW_DIGEST_REALM + 1;
676  pair->lvalue += 2;
677  pair->strvalue[1] = pair->lvalue;
678  pair->strvalue[pair->lvalue] = '\0';
679  pair->attribute = PW_DIGEST_ATTRIBUTES;
680  }
681 
682  pair->next = NULL;
683 
684  if (*first_pair == NULL)
685  {
686  *first_pair = pair;
687  }
688  else
689  {
690  link = *first_pair;
691  while (link->next != NULL)
692  {
693  link = link->next;
694  }
695  link->next = pair;
696  }
697 
698  mode = PARSE_MODE_NAME;
699  break;
700 
701  default:
702  mode = PARSE_MODE_NAME;
703  break;
704  }
705  }
706  return 0;
707 }

719 int rc_avpair_tostr (rc_handle const *rh, VALUE_PAIR *pair, char *name, int ln, char *value, int lv)
720 {
721  DICT_VALUE *dval;
722  char buffer[32];
723  struct in_addr inad;
724  unsigned char *ptr;
725  unsigned int pos;
726 
727  *name = *value = '\0';
728 
729  if (!pair || pair->name[0] == '\0') {
730  rc_log(LOG_ERR, "rc_avpair_tostr: pair is NULL or empty");
731  return -1;
732  }
733 
734  strlcpy(name, pair->name, (size_t) ln);
735 
736  switch (pair->type)
737  {
738  case PW_TYPE_STRING:
739  lv--;
740  pos = 0;
741  ptr = (unsigned char *) pair->strvalue;
742  if (pair->attribute == PW_DIGEST_ATTRIBUTES) {
743  pair->strvalue[*(ptr + 1)] = '\0';
744  ptr += 2;
745  }
746  while (*ptr != '\0')
747  {
748  if (!(isprint (*ptr)))
749  {
750  if (lv >= 4) {
751  snprintf (&value[pos], lv, "\\%03o", *ptr);
752  pos += 4;
753  lv -= 4;
754  } else {
755  break;
756  }
757  }
758  else
759  {
760  if (lv > 0) {
761  value[pos++] = *ptr;
762  lv--;
763  } else {
764  break;
765  }
766  }
767  ptr++;
768  }
769  if (lv > 0)
770  value[pos++] = 0;
771  else
772  value[pos-1] = 0;
773  break;
774 
775  case PW_TYPE_INTEGER:
776  dval = rc_dict_getval (rh, pair->lvalue, pair->name);
777  if (dval != NULL)
778  {
779  strlcpy(value, dval->name, (size_t) lv);
780  }
781  else
782  {
783  snprintf(value, lv, "%ld", (long int)pair->lvalue);
784  }
785  break;
786 
787  case PW_TYPE_IPADDR:
788  inad.s_addr = htonl(pair->lvalue);
789  strlcpy (value, inet_ntoa (inad), (size_t) lv);
790  break;
791 
792  case PW_TYPE_IPV6ADDR:
793  if (inet_ntop(AF_INET6, pair->strvalue, value, lv) == NULL)
794  return -1;
795  break;
796 
797  case PW_TYPE_IPV6PREFIX: {
798  uint8_t ip[16];
799  uint8_t txt[48];
800  if (pair->lvalue < 2)
801  return -1;
802 
803  memset(ip, 0, sizeof(ip));
804  memcpy(ip, pair->strvalue+2, pair->lvalue-2);
805 
806  if (inet_ntop(AF_INET6, ip, txt, sizeof(txt)) == NULL)
807  return -1;
808  snprintf(value, lv, "%s/%u", txt, (unsigned)pair->strvalue[1]);
809 
810  break;
811  }
812  case PW_TYPE_DATE:
813  strftime (value, lv, "%m/%d/%y %H:%M:%S",
814  gmtime ((time_t *) & pair->lvalue));
815  break;
816 
817  default:
818  rc_log(LOG_ERR, "rc_avpair_tostr: unknown attribute type %d", pair->type);
819  return -1;
820  break;
821  }
822 
823  return 0;
