67 int rc_aaa(rc_handle *rh, uint32_t client_port, VALUE_PAIR *send, VALUE_PAIR **received,
68  char *msg, int add_nas_port, int request_type)
69 {
70  SEND_DATA data;
71  VALUE_PAIR *adt_vp = NULL;
72  int result;
73  int i, skip_count;
74  SERVER *aaaserver;
75  int timeout = rc_conf_int(rh, "radius_timeout");
76  int retries = rc_conf_int(rh, "radius_retries");
77  int radius_deadtime = rc_conf_int(rh, "radius_deadtime");
78  double start_time = 0;
79  double now = 0;
80  time_t dtime;
81  rc_type type;
82 
83  if (rh->so_set == SOCKETS_TLS || rh->so_set == SOCKETS_DTLS ||
84  request_type != PW_ACCOUNTING_REQUEST) {
85  aaaserver = rc_conf_srv(rh, "authserver");
86  type = AUTH;
87  } else {
88  aaaserver = rc_conf_srv(rh, "acctserver");
89  type = ACCT;
90  }
91  if (aaaserver == NULL)
92  return ERROR_RC;
93 
94  data.send_pairs = send;
95  data.receive_pairs = NULL;
96 
97  if (add_nas_port != 0) {
98  /*
99  * Fill in NAS-Port
100  */
101  if (rc_avpair_add(rh, &(data.send_pairs), PW_NAS_PORT,
102  &client_port, 0, 0) == NULL)
103  return ERROR_RC;
104  }
105 
106  if (request_type == PW_ACCOUNTING_REQUEST) {
107  /*
108  * Fill in Acct-Delay-Time
109  */
110  dtime = 0;
111  now = rc_getctime();
112  adt_vp = rc_avpair_get(data.send_pairs, PW_ACCT_DELAY_TIME, 0);
113  if (adt_vp == NULL) {
114  adt_vp = rc_avpair_add(rh, &(data.send_pairs),
115  PW_ACCT_DELAY_TIME, &dtime, 0, 0);
116  if (adt_vp == NULL)
117  return ERROR_RC;
118  start_time = now;
119  } else {
120  start_time = now - adt_vp->lvalue;
121  }
122  }
123 
124  skip_count = 0;
125  result = ERROR_RC;
126  for (i=0; (i < aaaserver->max) && (result != OK_RC) && (result != BADRESP_RC)
127  ; i++, now = rc_getctime())
128  {
129  if (aaaserver->deadtime_ends[i] != -1 &&
130  aaaserver->deadtime_ends[i] > start_time) {
131  skip_count++;
132  continue;
133  }
134  if (data.receive_pairs != NULL) {
135  rc_avpair_free(data.receive_pairs);
136  data.receive_pairs = NULL;
137  }
138  rc_buildreq(rh, &data, request_type, aaaserver->name[i],
139  aaaserver->port[i], aaaserver->secret[i], timeout, retries);
140 
141  if (request_type == PW_ACCOUNTING_REQUEST) {
142  dtime = now - start_time;
143  rc_avpair_assign(adt_vp, &dtime, 0);
144  }
145 
146  result = rc_send_server (rh, &data, msg, type);
147  if (result == TIMEOUT_RC && radius_deadtime > 0)
148  aaaserver->deadtime_ends[i] = start_time + (double)radius_deadtime;
149  }
150  if (result == OK_RC || result == BADRESP_RC || skip_count == 0)
151  goto exit;
152 
153  result = ERROR_RC;
154  for (i=0; (i < aaaserver->max) && (result != OK_RC) && (result != BADRESP_RC)
155  ; i++)
156  {
157  if (aaaserver->deadtime_ends[i] == -1 ||
158  aaaserver->deadtime_ends[i] <= start_time) {
159  continue;
160  }
161  if (data.receive_pairs != NULL) {
162  rc_avpair_free(data.receive_pairs);
163  data.receive_pairs = NULL;
164  }
165  rc_buildreq(rh, &data, request_type, aaaserver->name[i],
166  aaaserver->port[i], aaaserver->secret[i], timeout, retries);
167 
168  if (request_type == PW_ACCOUNTING_REQUEST) {
169  dtime = rc_getctime() - start_time;
170  rc_avpair_assign(adt_vp, &dtime, 0);
171  }
172 
173  result = rc_send_server (rh, &data, msg, type);
174  if (result != TIMEOUT_RC)
175  aaaserver->deadtime_ends[i] = -1;
176  }
177 
178 exit:
179  if (request_type != PW_ACCOUNTING_REQUEST) {
180  *received = data.receive_pairs;
181  } else {
182  rc_avpair_free(data.receive_pairs);
183  }
184 
185  return result;
186 }

260 int rc_check(rc_handle *rh, char *host, char *secret, unsigned short port, char *msg)
261 {
262  SEND_DATA data;
263  int result;
264  uint32_t service_type;
265  int timeout = rc_conf_int(rh, "radius_timeout");
266  int retries = rc_conf_int(rh, "radius_retries");
267  rc_type type;
268 
269  data.send_pairs = data.receive_pairs = NULL;
270 
271  if (rh->so_set == SOCKETS_TLS || rh->so_set == SOCKETS_DTLS)
272  type = AUTH;
273  else
274  type = ACCT;
275 
276  /*
277  * Fill in Service-Type
278  */
279 
280  service_type = PW_ADMINISTRATIVE;
281  rc_avpair_add(rh, &(data.send_pairs), PW_SERVICE_TYPE, &service_type, 0, 0);
282 
283  rc_buildreq(rh, &data, PW_STATUS_SERVER, host, port, secret, timeout, retries);
284  result = rc_send_server (rh, &data, msg, type);
285 
286  rc_avpair_free(data.receive_pairs);
287 
288  return result;
